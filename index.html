async function stopJog() { if (!isConnected || !isJogging) return; isJogging = false; // Clear any ramp-up interval if (jogRampInterval) clearInterval(jogRampInterval); let currentSpeed = lastSentSpeed; // Smooth exponential decay while (Math.abs(currentSpeed) > 5) { currentSpeed *= 0.7; // deceleration curve await sendCommand(`JOGSLIDER:STOP:${Math.round(currentSpeed)}`); await new Promise(r => setTimeout(r, 20)); } // Final stop await sendCommand('JOGSLIDER:STOP:0'); lastSentSpeed = 0; log('Jog stopped smoothly'); }
